   119  // CreateAccount Create an account with AccountData data
   120  func (c *Client) CreateAccount(account *AccountData) (*AccountData, error) {
   121  	c.config.logDebug("Start CreateAccount")
   122  
   123  	if account == nil {
   124  		c.config.logError("No account data")
   125  		return nil, errors.New("no account data")
   126  	}
   127  
   128  	jsonStr, err := json.Marshal(accountDataDto{Data: account})
   129  	if err != nil {
   130  		c.config.logError("Json marshal account error : ", err.Error())
   131  		return nil, err
   132  	}
   133  
   134  	req, _ := http.NewRequest("POST", c.config.Address+apiBaseUri, bytes.NewBuffer(jsonStr))
   135  	req.Header.Add("Content-Type", "application/vnd.api+json")
   136  	resp, err := c.config.HttpClient.Do(req)
   137  	if err != nil {
   138  		c.config.logError("Http call API error : ", err.Error())
   139  		return nil, err
   140  	}
   141  
   142  	if resp.StatusCode != http.StatusCreated {
   143  		c.config.logError("create account error ", resp.StatusCode)
   144  		return nil, getDetailError(resp)
   145  	}
   146  
   147  	body, err := ioutil.ReadAll(resp.Body)
   148  	if err != nil {
   149  		return nil, err
   150  	}
   151  
   152  	var accountDto *accountDataDto
   153  	err = json.Unmarshal(body, &accountDto)
   154  	if err != nil {
   155  		c.config.logError("Failure: Unmarshal HTTP response error : " + err.Error())
   156  		return nil, errors.New("failure: Unmarshal HTTP response error : " + err.Error())
   157  	}
   158  
   159  	return accountDto.Data, nil
   160  }
